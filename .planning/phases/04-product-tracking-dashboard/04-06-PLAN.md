    ```typescript
    'use client'
    import { useActionState, useEffect, useRef } from 'react'
    import { useFormStatus } from 'react-dom'
    import { Loader2 } from 'lucide-react'
    import { toast } from 'sonner'
    import { Button } from '@/components/ui/button'
    import { Input } from '@/components/ui/input'
    import { Label } from '@/components/ui/label'
    import { useAuthModal } from '@/components/auth/AuthModalProvider'
    import { toastMessageForReason } from '@/lib/firecrawl/toast-messages'
    import { addProduct, type AddProductResult } from '@/actions/products'

    const PENDING_KEY = 'dealdrop:pending-add-url'

    type AddProductFormProps = Readonly<{
      authed: boolean
      variant?: 'inline' | 'dialog'
      onSubmitUrl?: (url: string) => void
      onSuccess?: () => void
    }>

    const initial: AddProductResult | null = null

    export function AddProductForm({ authed, variant = 'inline', onSubmitUrl, onSuccess }: AddProductFormProps) {
      const [state, formAction] = useActionState(addProduct, initial)
      const formRef = useRef<HTMLFormElement>(null)
      const { openAuthModal } = useAuthModal()

      // Toast dispatch on action state change.
      useEffect(() => {
        if (!state) return
        if (state.ok) {
          toast.success('Product added!')
          onSuccess?.()
          formRef.current?.reset()
        } else {
          toast.error(toastMessageForReason(state.reason))
        }
      }, [state, onSuccess])

      // D-03: on mount, if authed AND sessionStorage has a pending URL, auto-submit.
      useEffect(() => {
        if (!authed) return
        if (typeof window === 'undefined') return
        const pending = window.sessionStorage.getItem(PENDING_KEY)
        if (!pending) return
        window.sessionStorage.removeItem(PENDING_KEY)
        const form = formRef.current
        if (!form) return
        const input = form.elements.namedItem('url') as HTMLInputElement | null
        if (!input) return
        input.value = pending
        form.requestSubmit()
      }, [authed])

      function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
        if (authed) {
          // Optimistic callback fires for useOptimistic in ProductGrid (Plan 07)
          const url = (e.currentTarget.elements.namedItem('url') as HTMLInputElement).value
          onSubmitUrl?.(url)
          return
        }
        // Unauth branch — D-03
        e.preventDefault()
        const url = (e.currentTarget.elements.namedItem('url') as HTMLInputElement).value
        if (!url) return
        window.sessionStorage.setItem(PENDING_KEY, url)
        openAuthModal()
      }

      return (
        <form
          ref={formRef}
          action={formAction}
          onSubmit={handleSubmit}
          className="flex flex-col gap-2 w-full"
        >
          <Label htmlFor="add-product-url" className="sr-only">Product URL</Label>
          <div className="flex gap-2 w-full">
            <Input
              id="add-product-url"
              name="url"
              type="text"
              required
              placeholder="https://example.com/product"
              className="flex-1"
              autoComplete="off"
            />
            <SubmitButton />
          </div>
        </form>
      )
    }

    function SubmitButton() {
      const { pending } = useFormStatus()
      return (
        <Button type="submit" variant="default" disabled={pending}>
          {pending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
          Track
        </Button>
      )
    }
    ```

    Critical:
    - Line 1: `'use client'`
    - Input `type="text"` NOT `type="url"` (UI-SPEC: too strict for pasted URLs)
    - Storage key MUST be `'dealdrop:pending-add-url'` exactly (D-03)
    - Auto-submit guard: `typeof window === 'undefined'` protects the RSC-render pass
    - `form.requestSubmit()` triggers the form action (addProduct) properly — `form.submit()` skips validation + action

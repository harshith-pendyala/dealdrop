```typescript
import 'server-only'
import { createClient } from '@/lib/supabase/server'
import type { Tables } from '@/types/database'

export type PricePoint = {
  price: number
  currency: string
  checked_at: string
}

export type Product = Tables<'products'> & {
  price_history: PricePoint[]
}

export async function getUserProducts(): Promise<Product[]> {
  const supabase = await createClient()
  const { data, error } = await supabase
    .from('products')
    .select('*, price_history(price, currency, checked_at)')
    .order('created_at', { ascending: false })
    .order('checked_at', { ascending: true, referencedTable: 'price_history' })
  if (error) {
    console.error('getUserProducts: select failed', { err: error })
    return []
  }
  return (data ?? []) as Product[]
}
```

```typescript
import 'server-only'
// MUST be the first line — throws at bundle time if imported into a Client Component.
// Source: https://nextjs.org/docs/app/guides/data-security (server-only DAL pattern)

import { createClient } from '@/lib/supabase/server'
import type { Tables } from '@/types/database'

export type PricePoint = {
  price: number
  currency: string
  checked_at: string
}

export type Product = Tables<'products'> & {
  price_history: PricePoint[]
}

export async function getUserProducts(): Promise<Product[]> {
  const supabase = await createClient()
  // RLS policy products_select_own enforces user_id = auth.uid() on the outer table.
  // DB-06 ownership-chain policy on price_history applies automatically to the nested
  // embedded resource via PostgREST — no manual .eq('user_id', ...) needed. Per D-02.
  const { data, error } = await supabase
    .from('products')
    .select('*, price_history(price, currency, checked_at)')
    .order('created_at', { ascending: false })
    .order('checked_at', { ascending: true, referencedTable: 'price_history' })
  if (error) {
    console.error('getUserProducts: select failed', { err: error })
    return []  // fail-open to empty grid rather than crash the dashboard
  }
  return (data ?? []) as Product[]
}
```

**Preservation rules (enforced by the acceptance criteria):**
- Line 1 is exactly `import 'server-only'` (no blank lines, no shifted comments)
- `console.error('getUserProducts: select failed', { err: error })` wording is verbatim — it's what Wave 0's fail-open test spies for
- The fail-open `return []` pattern with its trailing comment stays verbatim
- `Tables<'products'>` is imported from `@/types/database`, NOT from `@supabase/supabase-js` or anywhere else

**Why the `as Product[]` cast is required:** Supabase's generated types do not auto-infer nested-select shapes from a select-string literal (05-RESEARCH.md §2 Type-Widening Strategy). The cast is safe because `PricePoint`'s three fields exactly mirror the selected columns from `Tables<'price_history'>`.

**Why two chained `.order()` calls (not `.order('...').order(...)` on separate lines or a single array)?** The PostgREST client treats each `.order()` as a separate ORDER clause; `referencedTable: 'price_history'` scopes the second one to the embedded resource. This is the current non-deprecated API (`foreignTable` is deprecated — do not use it).

**Do not add** `.limit(N)` — D-04 locked "no row limit" because the portfolio-demo scale makes pagination premature.
**Do not add** realtime subscriptions, client-side caching, or a parallel lazy-load DAL — all rejected in 05-CONTEXT.md §Deferred Ideas.

```tsx
'use client'
import {
  ResponsiveContainer,
  LineChart,
  Line,
  XAxis,
  YAxis,
  Tooltip,
} from 'recharts'
import type { PricePoint } from '@/lib/products/get-user-products'

type Props = Readonly<{ history: PricePoint[]; currency: string }>

export function yTickFormatter(value: number, currency: string): string {
  try {
    return new Intl.NumberFormat(undefined, {
      style: 'currency',
      currency,
      maximumFractionDigits: 0,
    }).format(value)
  } catch {
    return `${currency} ${Math.round(value)}`
  }
}

export function xTickFormatter(value: string): string {
  try {
    return new Intl.DateTimeFormat(undefined, {
      month: 'short',
      day: 'numeric',
    }).format(new Date(value))
  } catch {
    return value
  }
}

function fullPrice(amount: number, currency: string): string {
  try {
    return new Intl.NumberFormat(undefined, { style: 'currency', currency }).format(amount)
  } catch {
    return `${currency} ${amount.toFixed(2)}`
  }
}

function fullDate(value: string): string {
  try {
    return new Intl.DateTimeFormat(undefined, {
      year: 'numeric',
      month: 'long',
      day: 'numeric',
    }).format(new Date(value))
  } catch {
    return value
  }
}

type TooltipProps = {
  active?: boolean
  payload?: { value: number }[]
  label?: string
  currency: string
}

function PriceTooltip({ active, payload, label, currency }: TooltipProps) {
  if (!active || !payload?.length) return null
  const price = payload[0].value
  return (
    <div
      style={{
        background: 'var(--card)',
        border: '1px solid var(--border)',
        borderRadius: 6,
        padding: '8px 12px',
        color: 'var(--foreground)',
      }}
    >
      <p style={{ fontWeight: 600, fontSize: 14, margin: 0 }}>{fullDate(label ?? '')}</p>
      <p style={{ fontSize: 14, margin: 0 }}>{fullPrice(price, currency)}</p>
    </div>
  )
}

export function PriceChart({ history, currency }: Props) {
  // Defensive empty-state — TRACK-06 seeds the initial row atomically with
  // product creation, so this branch should not fire in practice. Matches the
  // existing ProductCard placeholder class pattern (min-h-[200px] bg-muted rounded-lg).
  if (history.length === 0) {
    return (
      <div className="min-h-[200px] bg-muted rounded-lg flex items-center justify-center">
        <p className="text-sm text-muted-foreground">No price history yet.</p>
      </div>
    )
  }

  return (
    <ResponsiveContainer width="100%" height={200}>
      <LineChart data={history} margin={{ top: 4, right: 8, bottom: 4, left: 0 }}>
        <XAxis
          dataKey="checked_at"
          tickFormatter={xTickFormatter}
          tick={{ fontSize: 14, fill: 'var(--muted-foreground)' }}
          tickLine={false}
          axisLine={false}
        />
        <YAxis
          tickFormatter={(value: number) => yTickFormatter(value, currency)}
          tick={{ fontSize: 14, fill: 'var(--muted-foreground)' }}
          domain={['auto', 'auto']}
          width={60}
          tickLine={false}
          axisLine={false}
        />
        <Tooltip content={<PriceTooltip currency={currency} />} />
        <Line
          type="monotone"
          dataKey="price"
          stroke="var(--primary)"
          strokeWidth={2}
          dot={history.length === 1 ? { r: 4, fill: 'var(--card)', stroke: 'var(--primary)', strokeWidth: 2 } : false}
          activeDot={{ r: 4, fill: 'var(--primary)' }}
        />
      </LineChart>
    </ResponsiveContainer>
  )
}
```

**Implementation notes (encoded in acceptance criteria):**

- **`'use client'` on line 1** — required by React 19 / Next.js 16 App Router for a module that uses Recharts (browser DOM measurements). ProductCard is already a client component (CLAUDE.md confirms `'use client'` at line 1 of ProductCard.tsx), so no new client boundary is created when ProductCard renders PriceChart in Wave 3.

- **Exported `xTickFormatter` + `yTickFormatter` as top-level functions** — enables Wave 0's formatter-unit-test assertions (`yTickFormatter(12, 'USD')` matches `/\$12|US\$12/`, `xTickFormatter('2026-04-20T00:00:00Z')` matches `/Apr\s*20/`) without rendering the chart. The YAxis `tickFormatter` prop binds `currency` inline: `(value) => yTickFormatter(value, currency)`.

- **Tooltip element-passing API (Risk 2)** — `content={<PriceTooltip currency={currency} />}` is the established Recharts 2.x→3.x-compatible pattern. If Wave 0's tests surface that Recharts 3.8.1 requires the render-function style, switch to `content={(props) => <PriceTooltip {...props} currency={currency} />}` and document the deviation in 05-02-SUMMARY.md. Both patterns satisfy the acceptance criteria.

- **Single-point dot** — `dot={history.length === 1 ? { r: 4, fill: 'var(--card)', stroke: 'var(--primary)', strokeWidth: 2 } : false}` — UI-SPEC §Color calls for the 1-point dot to have fill=`var(--card)` and stroke=`var(--primary)` so it reads as a distinct marker, not a solid disc. `activeDot` (hover) stays as a solid `var(--primary)` disc.

- **No CartesianGrid** — UI-SPEC §Color explicitly "No chart grid lines by default." Suppress unless a faint grid is added later for readability (deferred — not in Phase 5 scope).

- **No `<Legend>`** — single-series chart; legend would be redundant.

- **Domain `['auto', 'auto']`** — D-04 locked; lets Recharts auto-pad bounds so small price drops stay visible. Risk 3 acknowledges this may render oddly for single-point history with `min === max`; Wave 3's human-verify checkpoint catches visual regressions.

- **No `dynamic({ ssr: false })` import** — RESEARCH.md §5 confirms Recharts never executes on the Next.js server renderer for client components; no SSR shim needed.

**Do NOT** add:
- `import 'server-only'` — this is a client component
- State (`useState`), effects (`useEffect`), or transitions (`useTransition`)
- Any prop beyond `{ history, currency }` — UI-SPEC locked this shape
- A CartesianGrid, Legend, Brush, or any other Recharts sub-component not in the UI-SPEC skeleton
- Realtime subscriptions or caching — 05-CONTEXT.md §Deferred Ideas rejects them
